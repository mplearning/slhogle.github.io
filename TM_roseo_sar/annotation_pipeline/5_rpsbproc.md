
```python
protfams = open(project_path+'/prot_fams')

t = 'doms'
ecutoff = '1e-5'

for fam in protfams:
    fam = fam.rstrip()
    inloc = project_path+'/2nd_blasts/'+fam+'_76roseo.xml' 
    outloc = project_path+'/2nd_blasts/'+fam+'_76roseo.rpsbproc'
    
    subprocess.call(['rpsbproc',
                     '-i', inloc,
                     '-o', outloc,
                     '-t', t, '-e', ecutoff])
    
protfams.close()
```
